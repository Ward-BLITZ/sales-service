Supabase koppelen

Stap 1: Ga naar supabase.com → nieuw project → SQL Editor:
CREATE TABLE IF NOT EXISTS projects (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, data jsonb NOT NULL, created_at timestamptz DEFAULT now());
CREATE TABLE IF NOT EXISTS closed_projects (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, data jsonb NOT NULL, closed_at timestamptz DEFAULT now());
CREATE TABLE IF NOT EXISTS klanten (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, data jsonb NOT NULL, created_at timestamptz DEFAULT now());
CREATE TABLE IF NOT EXISTS settings (key text PRIMARY KEY, data jsonb NOT NULL);
ALTER TABLE projects ENABLE ROW LEVEL SECURITY;
ALTER TABLE closed_projects ENABLE ROW LEVEL SECURITY;
ALTER TABLE klanten ENABLE ROW LEVEL SECURITY;
ALTER TABLE settings ENABLE ROW LEVEL SECURITY;
CREATE POLICY "open" ON projects FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "open" ON closed_projects FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "open" ON klanten FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "open" ON settings FOR ALL USING (true) WITH CHECK (true);
Stap 2: Settings → API → kopieer Project URL en Publishable key
